-- 1. Bootstrap lazy.nvim - Установка если при запуске не обнаружено
local lazypath = vim.fn.stdpath("data") .. "/lazy/lazy.nvim"
if not (vim.uv or vim.loop).fs_stat(lazypath) then
  local lazyrepo = "https://github.com/folke/lazy.nvim.git"
  local out = vim.fn.system({
    "git",
    "clone",
    "--filter=blob:none",
    "--branch=stable",
    lazyrepo,
    lazypath })
  if vim.v.shell_error ~= 0 then
    vim.api.nvim_echo({
      { "Failed to clone lazy.nvim:\n", "ErrorMsg" },
      { out, "WarningMsg" },
      { "\nNeovim will run without plugins . Press any key...", "None" },
    }, true, {})
    vim.fn.getchar()
    -- os.exit(1)
  end
end

-- 2. Запускаем setup только если папка существует
if vim.uv.fs_stat(lazypath) then
  vim.opt.rtp:prepend(lazypath)
  require("lazy").setup({})
else
  vim.api.nvim_echo({{ "Directory lazy.nvim not found. Plugins didn't load.", "WarningMsg" }}, true, {})
end
-- Проверить :checkhealth lazy

-- 3. Указание lazi.nvim проверить в директории 'lua/plugins'
require("lazy").setup("plugins", {
  change_detection = { notify = false }, -- Не уведомлять об изменениях конфига
  rocks = {
      enabled = false, -- Отключить LuaRocks (уберет ошибки hererocks)
  },
})
